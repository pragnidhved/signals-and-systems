\textbf{Question 10.5.2.8:}\\
An AP consists of $50$ terms of which $3^{rd}$ term is $12$ and the last term is $106$. Find the $29^{th}$ term.\\
\solution 
\begin{table}[H]
\centering
\begin{tabular}{|c|c|c|}\hline
\textbf{Parameter} & \textbf{Value} & \textbf{description}\\ \hline
$x(2)$ & $12$ & Third term\\ \hline
$x(49)$ & $106$ & Last term\\ \hline
$x(0)$ & $$ & First term \\ \hline
$d$ & $$ & Common difference\\ \hline
$x(n)$ & $(x(0)+nd)u(n)$ & general term \\ \hline
\end{tabular}
\caption{Input parameters}
\label{tab:table1}
\end{table}
\begin{align}
\myvec{x(2) \\ x(49)}
&=
\myvec{1 & 2 \\ 1 & 49}
\myvec{x(0) \\ d}
\\[5pt]
\myvec{12 \\ 106}
&=
\myvec{x(0) & 2d \\ x(0) & 49d}
\quad R_2\rightarrow R_2-R_1\\[5pt]
\myvec{12 \\ 94}
&=
\myvec{x(0) & 2d \\ 0 & 47d}
\end{align}
\begin{enumerate}
\item From column 2 we get
\begin{align}
\implies d=2
\label{eq:4}
\end{align}
\item Substituting \eqref{eq:4} in column 1, we get
\begin{align}
\implies x(0)=8
\end{align}
\item From the \tabref{tab:table1} :
\begin{align}
x(n)&=(x(0)+nd)u(n)\\
\implies x(n)&=(8+2n)u(n)
\end{align}
\item Finding $x(28)$ :
\begin{align}
x(28)&=x(0)+28(2)\\
\implies x(28)&=64
\end{align}
\item Z-transform :
\end{enumerate}\\
\begin{equation}
\implies X(z)=\frac{8-6z^{-1}}{(1-z^{-1})^2} \quad \abs{z}>1
\end{equation}

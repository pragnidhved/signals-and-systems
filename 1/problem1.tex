\textbf{Question 10.5.2.8:}\\
An AP consists of $50$ terms of which $3^{rd}$ term is $12$ and the last term is $106$. Find the $29^{th}$ term.\\
\textbf{Solution :}\\
\begin{table}[H]
\centering
\begin{tabular}{|c|c|c|}\hline
$x_3$ & $x_{50}$ & $x_{n}$\\ \hline
$12$ & $106$ & $a_0+nd$\\ \hline
\end{tabular}
\caption{Given inputs}
\end{table}
For an AP
\begin{equation}
x_n-x_{n-1}=d
\end{equation}
Given \\
\begin{equation}
 x_0+3d=12
 \end{equation}
 Given \\
 \begin{equation}
x_0+50d=106
 \end{equation}
 $(2)-(1)$
 \begin{align}
 x_0+50d-(x_0+3d)&=106-12\\
 47d&=94\\
 d&=2
 \end{align}
 Substituting $d$ in (1)
 \begin{align}
 x_0+3(2)&=12\\
 x_0&=6
 \end{align}
 We know that
 \begin{equation}
 x_{29}=x_0+29d
 \end{equation}
 By (8) and (6) in (9) 
 \begin{align}
 x_{29}&=6+29(2)\\
 x_{29}&=6+58\\
 x_{29}&=64
 \end{align}
